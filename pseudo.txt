Algoritmo Cumpleaños_N_Personas
    Escribir "Ingrese cantidad de personas:"
    Leer n
    Dimension nombres[n], dias[n] // Arreglos para guardar datos
    
    // Ciclo para capturar datos
    Para i <- 1 Hasta n Hacer
        Escribir "Nombre ", i, ":"
        Leer nombres[i]
        Escribir "Día de cumple (1-31):"
        Leer dias[i]
    FinPara
    
    // Ciclos anidados para comparar
    Para i <- 1 Hasta n-1 Hacer
        Para j <- i + 1 Hasta n Hacer
            Si dias[i] = dias[j] Entonces
                Escribir nombres[i], " y ", nombres[j], " coinciden."
            Sino
                diff <- abs(dias[i] - dias[j])
                Si diff <= 3 Entonces
                    Escribir nombres[i], " y ", nombres[j], " son cercanos."
                FinSi
            FinSi
        FinPara
    FinPara
FinAlgoritmo
