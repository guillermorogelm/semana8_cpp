Algoritmo CumpleañosSimple
    // Entrada de datos persona 1
    Escribir "Nombre 1:"
    Leer nom1
    Escribir "Día (1-31):"
    Leer d1
    
    // Entrada de datos persona 2
    Escribir "Nombre 2:"
    Leer nom2
    Escribir "Día (1-31):"
    Leer d2
    
    // Proceso de comparación
    Si d1 = d2 Entonces
        Escribir nom1, " y ", nom2, " cumplen el mismo día."
    Sino
        Si d1 > d2 Entonces diff <- d1 - d2 Sino diff <- d2 - d1 FinSi
        
        Si diff <= 3 Entonces
            Escribir "Tienen fechas cercanas. Diferencia: ", diff, " días."
        Sino
            Escribir "No cumplen días cercanos."
        FinSi
    FinSi
FinAlgoritmo
